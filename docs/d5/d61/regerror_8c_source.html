196 static int to_ascii(OnigEncoding enc, UChar *s, UChar *end,
197  UChar buf[], int buf_size, int *is_over)
198 {
199  int len;
200  UChar *p;
201  OnigCodePoint code;
202 
203  if (ONIGENC_MBC_MINLEN(enc) > 1) {
204  p = s;
205  len = 0;
206  while (p < end) {
207  code = ONIGENC_MBC_TO_CODE(enc, p, end);
208  if (code >= 0x80) {
209  if (code > 0xffff && len + 10 <= buf_size) {
210  sprint_byte_with_x((char*)(&(buf[len])), (unsigned int)(code >> 24));
211  sprint_byte((char*)(&(buf[len+4])), (unsigned int)(code >> 16));
212  sprint_byte((char*)(&(buf[len+6])), (unsigned int)(code >> 8));
213  sprint_byte((char*)(&(buf[len+8])), (unsigned int)code);
214  len += 10;
215  }
216  else if (len + 6 <= buf_size) {
217  sprint_byte_with_x((char*)(&(buf[len])), (unsigned int)(code >> 8));
218  sprint_byte((char*)(&(buf[len+4])), (unsigned int)code);
219  len += 6;
220  }
221  else {
222  break;
223  }
224  }
225  else {
226  buf[len++] = (UChar )code;
227  }
228 
229  p += enclen(enc, p, end);
230  if (len >= buf_size) break;
231  }
232 
233  *is_over = ((p < end) ? 1 : 0);
234  }
235  else {
236  len = (int )MIN((end - s), buf_size);
237  xmemcpy(buf, s, (size_t )len);
238  *is_over = ((buf_size < (end - s)) ? 1 : 0);
239  }
240 
241  return len;
242 }

248 extern int
249 onig_error_code_to_str(UChar* s, OnigPosition code, ...)
250 {
251  UChar *p, *q;
252  OnigErrorInfo* einfo;
253  size_t len;
254  int is_over;
255  UChar parbuf[MAX_ERROR_PAR_LEN];
256  va_list vargs;
257 
258  va_start(vargs, code);
259 
260  switch (code) {
261  case ONIGERR_UNDEFINED_NAME_REFERENCE:
262  case ONIGERR_UNDEFINED_GROUP_REFERENCE:
263  case ONIGERR_MULTIPLEX_DEFINED_NAME:
264  case ONIGERR_MULTIPLEX_DEFINITION_NAME_CALL:
265  case ONIGERR_INVALID_GROUP_NAME:
266  case ONIGERR_INVALID_CHAR_IN_GROUP_NAME:
267  case ONIGERR_INVALID_CHAR_PROPERTY_NAME:
268  einfo = va_arg(vargs, OnigErrorInfo*);
269  len = to_ascii(einfo->enc, einfo->par, einfo->par_end,
270  parbuf, MAX_ERROR_PAR_LEN - 3, &is_over);
271  q = onig_error_code_to_format(code);
272  p = s;
273  while (*q != '\0') {
274  if (*q == '%') {
275  q++;
276  if (*q == 'n') { /* '%n': name */
277  xmemcpy(p, parbuf, len);
278  p += len;
279  if (is_over != 0) {
280  xmemcpy(p, "...", 3);
281  p += 3;
282  }
283  q++;
284  }
285  else
286  goto normal_char;
287  }
288  else {
289  normal_char:
290  *p++ = *q++;
291  }
292  }
293  *p = '\0';
294  len = p - s;
295  break;
296 
297  default:
298  q = onig_error_code_to_format(code);
299  if (q) {
300  len = onigenc_str_bytelen_null(ONIG_ENCODING_ASCII, q);
301  xmemcpy(s, q, len);
302  } else {
303  len = 0;
304  }
305  s[len] = '\0';
306  break;
307  }
308 
309  va_end(vargs);
310  return (int )len;
311 }

313 void
314 onig_vsnprintf_with_pattern(UChar buf[], int bufsize, OnigEncoding enc,
315  UChar* pat, UChar* pat_end, const UChar *fmt, va_list args)
316 {
317  size_t need;
318  int n, len;
319  UChar *p, *s, *bp;
320  UChar bs[6];
321 
322  n = xvsnprintf((char* )buf, bufsize, (const char* )fmt, args);
323 
324  need = (pat_end - pat) * 4 + 4;
325 
326  if (n + need < (size_t )bufsize) {
327  static const char sep[] = ": /";
328  memcpy((char* )buf + n, sep, sizeof(sep));
329  s = buf + onigenc_str_bytelen_null(ONIG_ENCODING_ASCII, buf);
330 
331  p = pat;
332  while (p < pat_end) {
333  if (ONIGENC_IS_MBC_HEAD(enc, p, pat_end)) {
334  len = enclen(enc, p, pat_end);
335  if (ONIGENC_MBC_MINLEN(enc) == 1) {
336  while (len-- > 0) *s++ = *p++;
337  }
338  else { /* for UTF16/32 */
339  int blen;
340 
341  while (len-- > 0) {
342  sprint_byte_with_x((char* )bs, (unsigned int )(*p++));
343  blen = onigenc_str_bytelen_null(ONIG_ENCODING_ASCII, bs);
344  bp = bs;
345  while (blen-- > 0) *s++ = *bp++;
346  }
347  }
348  }
349  else if (*p == '\\') {
350  *s++ = *p++;
351  len = enclen(enc, p, pat_end);
352  while (len-- > 0) *s++ = *p++;
353  }
354  else if (*p == '/') {
355  *s++ = (unsigned char )'\\';
356  *s++ = *p++;
357  }
358  else if (!ONIGENC_IS_CODE_PRINT(enc, *p) &&
359  !ONIGENC_IS_CODE_SPACE(enc, *p)) {
360  sprint_byte_with_x((char* )bs, (unsigned int )(*p++));
361  len = onigenc_str_bytelen_null(ONIG_ENCODING_ASCII, bs);
362  bp = bs;
363  while (len-- > 0) *s++ = *bp++;
364  }
365  else {
366  *s++ = *p++;
367  }
368  }
369 
370  *s++ = '/';
371  *s = '\0';
372  }
373 }
