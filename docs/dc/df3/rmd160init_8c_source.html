22 /*
23  * Document-class: Digest::RMD160 < Digest::Base
24  * A class for calculating message digests using RIPEMD-160
25  * cryptographic hash function, designed by Hans Dobbertin, Antoon
26  * Bosselaers, and Bart Preneel.
27  *
28  * RMD160 calculates a digest of 160 bits (20 bytes).
29  *
30  * == Examples
31  * require 'digest'
32  *
33  * # Compute a complete digest
34  * Digest::RMD160.hexdigest 'abc' #=> "8eb208f7..."
35  *
36  * # Compute digest by chunks
37  * rmd160 = Digest::RMD160.new # =>#<Digest::RMD160>
38  * rmd160.update "ab"
39  * rmd160 << "c" # alias for #update
40  * rmd160.hexdigest # => "8eb208f7..."
41  *
42  * # Use the same object to compute another digest
43  * rmd160.reset
44  * rmd160 << "message"
45  * rmd160.hexdigest # => "1dddbe1b..."
46  */
