38 {
39  SSL_SESSION *ctx = NULL;

41  if (RDATA(self)->data)

46

51  } else {

53 
54  ctx = PEM_read_bio_SSL_SESSION(in, NULL, NULL, NULL);

59  }
60

65  }

67  /* should not happen */

69  ossl_raise(eSSLSession, "ctx not set - internal error");

74 }
